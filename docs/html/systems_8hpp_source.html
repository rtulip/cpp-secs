177  void run(system_data data)
178  {
179  auto paddle_pos = std::get<0>(data);
180  auto paddle_rect = std::get<1>(data);
181  auto paddle_side = std::get<2>(data);
182  auto world = std::get<3>(data);
183 
184  // Do a fetch for the 'Balls' in the run function.
185  auto balls = world->world()->safe_fetch<pc::Position, pc::Rectangle, pc::Velocity, pc::Ball>();
186  for (auto ball : balls)
187  {
188 
189  auto ball_pos = std::get<0>(ball);
190  auto ball_rect = std::get<1>(ball);
191  auto ball_vel = std::get<2>(ball);
192  auto ball_speed = std::get<3>(ball);
193 
194  if (overlap(paddle_pos, paddle_rect, ball_pos, ball_rect))
195  {
196  float paddle_top = paddle_pos->y;
197  float paddle_left = paddle_pos->x;
198  float paddle_right = paddle_pos->x + paddle_rect->width;
199  float paddle_bot = paddle_pos->y - paddle_rect->height;
200  float ball_top = ball_pos->y;
201  float ball_bot = ball_pos->y - ball_rect->height;
202  float ball_left = ball_pos->x;
203  float ball_right = ball_pos->x + ball_rect->width;
204 
205  float ball_center = ball_pos->y - (ball_rect->height / 2.0);
206  float paddle_y_center = paddle_pos->y - (paddle_rect->height / 2.0);
207  float ball_y_relative_to_paddle = ball_center - paddle_y_center;
208  float ball_y_relative_to_paddle_normalized = ball_y_relative_to_paddle / (paddle_rect->height / 2.0);
209 
210  float ball_new_vel_dx_sign = (ball_vel->dx > 0 ? -1.0 : 1.0);
211 
212  // Correct case where ball center is above paddle
213  if (std::abs(ball_y_relative_to_paddle_normalized) > 1.0)
214  ball_y_relative_to_paddle_normalized = (ball_y_relative_to_paddle_normalized < 0 ? -1.0 : 1.0);
215 
216  // Fix the collision
217  while (overlap(paddle_pos, paddle_rect, ball_pos, ball_rect))
218  {
219  ball_pos->x -= ball_vel->dx;
220  ball_pos->y -= ball_vel->dy;
221  }
222 
223  // Set the new velocity
224  float new_x_vel = ball_speed->speed * cos(MAX_BOUNCE_ANGLE * ball_y_relative_to_paddle_normalized) * ball_new_vel_dx_sign;
225  float new_y_vel = ball_speed->speed * sin(MAX_BOUNCE_ANGLE * ball_y_relative_to_paddle_normalized);
226  *ball_vel = {new_x_vel, new_y_vel};
227  }
228  }
229  }
