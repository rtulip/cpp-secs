145  void DispatcherContainerBuilder::done()
146  {
147  while (!this->systems.empty())
148  {
149  // Add empty stage
150  ecs::dispatch::DispatcherStage stage;
151 
152  // Find all keys with out degree 0
153  std::vector<std::string> in_degree_zero_keys;
154  for (auto key : this->counts)
155  {
156  if (key.second == 0)
157  in_degree_zero_keys.push_back(key.first);
158  }
159 
160  // Remove the keys from counts.
161  for (auto key : in_degree_zero_keys)
162  this->counts.erase(key);
163 
164  // Check for cycles - Kept as a failsafe.
165  if (in_degree_zero_keys.size() == 0 && !this->systems.empty())
166  throw std::runtime_error("Detected cycle in dependency graph!");
167 
168  // Remove the systems with in degree 0
169  for (auto key : in_degree_zero_keys)
170  {
171  // Decrease the in degree of each element pointed by key
172  for (auto edge : this->edges[key])
173  {
174  this->counts[edge]--;
175  }
176 
177  // Remove the key
178  this->edges.erase(key);
179  // Remove the system from the graph, and add it to the stage.
180  stage.push_back(this->systems[key]);
181  this->systems.erase(key);
182  }
183 
184  // Insert the stage into the dependency graph.
185  container_ref->insert(container_ref->begin(), stage);
186  }
187  }

651  void WorldResource::remove_entity_component(Entity *e)
652  {
653  // Get the component id of T.
654  size_t cid = this->world_ptr->get_cid<T>();
655 
656  size_t ENTITY_CID = this->world_ptr->get_cid<Entity>();
657  if (cid == ENTITY_CID)
658  throw std::runtime_error("Cannot remove the Entity component from an Entity!");
659 
660  // If the entity doesn't have a T component, throw a runtime error. This is to
661  // help ensure that a system is only operating on the entity it's working with.
662  if (!e->has_component(cid))
663  throw std::runtime_error("Cannot remove a component from an entity if it doesn't have it.");
664 
665  // Get the index of the component to be removed.
666  size_t entity_component_index = e->get_component(cid);
667 
668  // Fetch the RegistryNode for the component
669  RegistryNode *node_ptr = this->world_ptr->find<T>();
670 
671  // Create a lambda function to delete the component instance, and remove the
672  // entity's knowledge of the component.
673  auto f = [node_ptr, entity_component_index, e, cid]() {
674  e->remove_component(cid); // Forget the component
675  node_ptr->erase<T>(entity_component_index); // Delete the component
676  };
677 
678  // Add the lambda function to be called later paired with the component index.
679  this->remove_functions.push_back(std::make_tuple(cid, entity_component_index, std::move(f)));

818  void WorldResource::merge()
819  {
820  for (auto &f : this->add_functions)
821  f();
822 
823  this->add_functions.clear();
824 
825  // If nothing to remove, return early for faster execution.
826  if (this->remove_functions.size() == 0)
827  return;
828 
842  auto function_order_sort = [](std::tuple<size_t, size_t, std::function<void()>> lhs, std::tuple<size_t, size_t, std::function<void()>> rhs) {
843  if (std::get<0>(lhs) > std::get<0>(rhs))
844  return true;
845  return std::get<1>(lhs) > std::get<1>(rhs);
846  };
847 
848  // Sort the list as described above.
849  std::sort(this->remove_functions.begin(), this->remove_functions.end(), function_order_sort);
850 
851  // Remove all the components staged for removal.
852  for (auto tuple : this->remove_functions)
853  {
854  size_t cid = std::get<0>(tuple);
855  size_t idx = std::get<1>(tuple);
856  auto f = std::get<2>(tuple);
857  f();
858  }
859 
860  // Adjust indicies of every Entity effected.
861  RegistryNode *entity_node = this->world_ptr->find<Entity>();
862  for (auto &e : *(entity_node->iter<Entity>()))
863  {
864  for (auto tuple : this->remove_functions)
865  {
866  size_t cid = std::get<0>(tuple);
867  size_t idx = std::get<1>(tuple);
868  // The component MUST be valid, otherwise it may have been removed already.
869  if (e.has_valid_component(cid))
870  {
871  size_t e_idx = e.get_component(cid);
872 
873  if (e_idx > idx)
874  {
875  e.decrement_component(cid);
876  }
877  }
878  }
879  }
880  // Empty the lists for the next systems.
881  this->remove_functions.clear();
882  }

903  {
904  RegistryNode *world_res_node = this->find<WorldResource>();
905  WorldResource *world_res = world_res_node->get<WorldResource>(0);
906  for (auto &stage : this->systems)
907  {
908  if (stage.size() > 1)
909  {
920  std::vector<std::thread> threads;
921  for (auto sys : stage)
922  {
923  auto lambda_f = [sys](ecs::world::World *w) {
924  sys->exec(w);
925  };
926  std::thread t(lambda_f, this);
927  threads.push_back(std::move(t));
928  }
929 
930  // Wait for each thread to join before moving to the next stage.
931  for (int i = 0; i < threads.size(); i++)
932  {
933  threads.at(i).join();
934  }
935  }
936  else // If only one system in a stage, run in the main thread.
937  {
938  for (auto sys : stage)
939  sys->exec(this);
940  }
941  // Perform any removals required now that all threads have joined.
942  world_res->merge();
943  }
944  }
