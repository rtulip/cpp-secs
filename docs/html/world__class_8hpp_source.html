132  void DispatcherContainerBuilder::done()
133  {
134  while (!this->systems.empty())
135  {
136  // Add empty stage
137  ecs::dispatch::DispatcherStage stage;
138 
139  // Find all keys with out degree 0
140  std::vector<std::string> in_degree_zero_keys;
141  for (auto key : this->counts)
142  {
143  if (key.second == 0)
144  in_degree_zero_keys.push_back(key.first);
145  }
146 
147  // Remove the keys from counts.
148  for (auto key : in_degree_zero_keys)
149  this->counts.erase(key);
150 
151  // Check for cycles - Kept as a failsafe.
152  if (in_degree_zero_keys.size() == 0 && !this->systems.empty())
153  throw std::runtime_error("Detected cycle in dependency graph!");
154 
155  // Remove the systems with in degree 0
156  for (auto key : in_degree_zero_keys)
157  {
158  // Decrease the in degree of each element pointed by key
159  for (auto edge : this->edges[key])
160  {
161  this->counts[edge]--;
162  }
163 
164  // Remove the key
165  this->edges.erase(key);
166  // Remove the system from the graph, and add it to the stage.
167  stage.push_back(this->systems[key]);
168  this->systems.erase(key);
169  }
170 
171  // Insert the stage into the dependency graph.
172  container_ref->insert(container_ref->begin(), stage);
173  }
174  }

638  void WorldResource::remove_entity_component(Entity *e)
639  {
640  // Get the component id of T.
641  size_t cid = this->world_ptr->get_cid<T>();
642 
643  size_t ENTITY_CID = this->world_ptr->get_cid<Entity>();
644  if (cid == ENTITY_CID)
645  throw std::runtime_error("Cannot remove the Entity component from an Entity!");
646 
647  // If the entity doesn't have a T component, throw a runtime error. This is to
648  // help ensure that a system is only operating on the entity it's working with.
649  if (!e->has_component(cid))
650  throw std::runtime_error("Cannot remove a component from an entity if it doesn't have it.");
651 
652  // Get the index of the component to be removed.
653  size_t entity_component_index = e->get_component(cid);
654 
655  // Fetch the RegistryNode for the component
656  RegistryNode *node_ptr = this->world_ptr->find<T>();
657 
658  // Create a lambda function to delete the component instance, and remove the
659  // entity's knowledge of the component.
660  auto f = [node_ptr, entity_component_index, e, cid]() {
661  e->remove_component(cid); // Forget the component
662  node_ptr->erase<T>(entity_component_index); // Delete the component
663  };
664 
665  // Add the lambda function to be called later paired with the component index.
666  this->remove_functions.push_back(std::make_tuple(cid, entity_component_index, std::move(f)));

805  void WorldResource::merge()
806  {
807  for (auto &f : this->add_functions)
808  f();
809 
810  this->add_functions.clear();
811 
812  // If nothing to remove, return early for faster execution.
813  if (this->remove_functions.size() == 0)
814  return;
815 
829  auto function_order_sort = [](std::tuple<size_t, size_t, std::function<void()>> lhs, std::tuple<size_t, size_t, std::function<void()>> rhs) {
830  if (std::get<0>(lhs) > std::get<0>(rhs))
831  return true;
832  return std::get<1>(lhs) > std::get<1>(rhs);
833  };
834 
835  // Sort the list as described above.
836  std::sort(this->remove_functions.begin(), this->remove_functions.end(), function_order_sort);
837 
838  // Remove all the components staged for removal.
839  for (auto tuple : this->remove_functions)
840  {
841  size_t cid = std::get<0>(tuple);
842  size_t idx = std::get<1>(tuple);
843  auto f = std::get<2>(tuple);
844  f();
845  }
846 
847  // Adjust indicies of every Entity effected.
848  RegistryNode *entity_node = this->world_ptr->find<Entity>();
849  for (auto &e : *(entity_node->iter<Entity>()))
850  {
851  for (auto tuple : this->remove_functions)
852  {
853  size_t cid = std::get<0>(tuple);
854  size_t idx = std::get<1>(tuple);
855  // The component MUST be valid, otherwise it may have been removed already.
856  if (e.has_valid_component(cid))
857  {
858  size_t e_idx = e.get_component(cid);
859 
860  if (e_idx > idx)
861  {
862  e.decrement_component(cid);
863  }
864  }
865  }
866  }
867  // Empty the lists for the next systems.
868  this->remove_functions.clear();
869  }

907  {
908  RegistryNode *world_res_node = this->find<WorldResource>();
909  WorldResource *world_res = world_res_node->get<WorldResource>(0);
910  for (auto &stage : this->systems)
911  {
912  if (stage.size() > 1)
913  {
924  std::vector<std::thread> threads;
925  for (auto sys : stage)
926  {
927  auto lambda_f = [sys](ecs::world::World *w) {
928  sys->exec(w);
929  };
930  std::thread t(lambda_f, this);
931  threads.push_back(std::move(t));
932  }
933 
934  // Wait for each thread to join before moving to the next stage.
935  for (int i = 0; i < threads.size(); i++)
936  {
937  threads.at(i).join();
938  }
939  }
940  else // If only one system in a stage, run in the main thread.
941  {
942  for (auto sys : stage)
943  sys->exec(this);
944  }
945  // Perform any removals required now that all threads have joined.
946  world_res->merge();
947  }
948  }
